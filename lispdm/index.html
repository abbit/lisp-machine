LispDM is a Scheme interpreter. Currently, it supports only a subset of Scheme.

use lispdm::Engine;
let mut engine = Engine::default();

let result = engine.eval::<i64>("(+ 1 2)").unwrap();
assert_eq!(result, Ok(3));
